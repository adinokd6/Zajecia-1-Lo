#jak przyjąć argumenty gdy nie wiemy ile ich będzie?
def funkcja(*nati):
  iloczyn = 1
  for arg in nati:
    iloczyn = iloczyn * arg
  print(iloczyn)
  
funkcja()
funkcja(1,2,3,4)
